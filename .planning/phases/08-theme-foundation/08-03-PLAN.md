**Part A: Update model.go for 't' key handler**

1. Add state to Model for picker mode:
   ```go
   showPicker bool  // True when showing theme picker
   picker     ThemePickerModel
   ```

2. In Update(), add case for ChangeTheme key (after other key handlers):
   ```go
   case key.Matches(msg, m.keys.ChangeTheme):
       m.showPicker = true
       m.picker = NewThemePicker()
       // Pre-select current theme in picker
       for i, name := range theme.AllThemes() {
           if name == m.theme.Name || strings.ToLower(m.theme.Name) == name {
               m.picker.selectedIdx = i
               break
           }
       }
       return m, nil
   ```

3. In Update(), at the START of the switch (before other message handling), add picker delegation when showPicker is true:
   ```go
   if m.showPicker {
       // Delegate to picker
       switch msg := msg.(type) {
       case tea.WindowSizeMsg:
           m.picker.width = msg.Width
           m.picker.height = msg.Height
           m.width = msg.Width
           m.height = msg.Height
       case tea.KeyMsg:
           var cmd tea.Cmd
           newPicker, _ := m.picker.Update(msg)
           m.picker = newPicker.(ThemePickerModel)
           if m.picker.confirmed || m.picker.cancelled {
               // Picker done - apply theme
               themeName := m.picker.SelectedTheme()
               m.theme = theme.GetTheme(themeName)
               m.showPicker = false
               // Update spinner style with new theme
               m.spinner.Style = lipgloss.NewStyle().Foreground(m.theme.Primary)
               return m, cmd
           }
           return m, cmd
       }
       return m, nil
   }
   ```

4. In View(), at the START (after quitting check), handle picker view:
   ```go
   if m.showPicker {
       return m.picker.View()
   }
   ```

**Part B: Update main.go for first-run detection**

1. Add import for theme package:
   `"github.com/b00y0h/wakadash/internal/theme"`

2. After config loading, before creating Model, detect first run:
   ```go
   // Check if first run (no theme configured)
   themeName, _ := theme.LoadThemeFromConfig()
   isFirstRun := themeName == ""
   ```

3. If first run, launch picker first:
   ```go
   if isFirstRun {
       // Show theme picker on first run
       picker := tui.NewThemePicker()
       pickerProgram := tea.NewProgram(picker, tea.WithAltScreen())
       finalModel, err := pickerProgram.Run()
       if err != nil {
           log.Fatalf("theme picker error: %v", err)
       }

       // Get selected theme from picker
       pickerResult := finalModel.(tui.ThemePickerModel)
       selectedTheme := pickerResult.SelectedTheme()

       // Save theme (picker already saved, but ensure it's saved)
       theme.SaveThemeToConfig(selectedTheme)
   }
   ```

4. Now create Model as usual (it will load the saved theme):
   ```go
   model := tui.NewModel(client, rangeStr, refreshInterval)
   p := tea.NewProgram(model, tea.WithAltScreen())
   ```
